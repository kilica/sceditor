249 		/**
250 		 * Creates the toolbar and appends it to the container
251 		 * @private
252 		 */
253 		initToolBar = function () {
254 			var	group, buttons,
255 				accessibilityName,
256 				button, i, buttonClick,
257 				groups = base.options.toolbar.split("|");
258 			
259 			buttonClick = function (e) {
260 				e.preventDefault();
261 				
262 				if($(this).hasClass('disabled'))
263 					return;
264 				
265 				handleCommand($(this), base.commands[$(this).data("sceditor-command")]);
266 				return false;
267 			};
268 
269 			$toolbar = $('<div class="sceditor-toolbar" />');
270 			for (i=0; i < groups.length; i++) {
271 				group   = $('<div class="sceditor-group" />');
272 				buttons = groups[i].split(",");
273 
274 				for (var x=0; x < buttons.length; x++) {
275 					// the button must be a valid command otherwise ignore it
276 					if(!base.commands.hasOwnProperty(buttons[x]))
277 						continue;
278 
279 					accessibilityName = base.commands[buttons[x]].tooltip ? base._(base.commands[buttons[x]].tooltip) : buttons[x];
280 					
281 					button = $('<a class="sceditor-button sceditor-button-' + buttons[x] +
282 						' " unselectable="on"><div unselectable="on">' + accessibilityName + '</div></a>');
283 
284 					if(base.commands[buttons[x]].hasOwnProperty("tooltip"))
285 						button.attr('title', base._(base.commands[buttons[x]].tooltip));
286 						
287 					if(base.commands[buttons[x]].exec)
288 						button.data('sceditor-wysiwygmode', true);
289 					else
290 						button.addClass('disabled');
291 						
292 					if(base.commands[buttons[x]].txtExec)
293 						button.data('sceditor-txtmode', true);
294 
295 					// add the click handler for the button
296 					button.data("sceditor-command", buttons[x]);
297 					button.click(buttonClick);
298 
299 					group.append(button);
300 				}
301 				$toolbar.append(group);
302 			}
303 
304 			// append the toolbar to the toolbarContainer option if given
305 			if(base.options.toolbarContainer === null)
306 				$editorContainer.append($toolbar);
307 			else
308 				$(base.options.toolbarContainer).append($toolbar);
309 		};

316 			var	doc	= wysiwygEditor.contentWindow.document,
317 				body	= doc.body, rng;
318 
319 			if(!doc.createRange)
320 				return base.focus();
321 			
322 			if(!body.firstChild)
323 				return;
324 			
325 			rng = doc.createRange();
326 			rng.setStart(body.firstChild, 0);
327 			rng.setEnd(body.firstChild, 0);

329 			rangeHelper.selectRange(rng);
330 			body.focus();

516 		/**
517 		 * Creates the resizer.
518 		 * @private
519 		 */
520 		initResize = function () {
521 			var	$grip		= $('<div class="sceditor-grip" />'),
522 				// cover is used to cover the editor iframe so document still gets mouse move events
523 				$cover		= $('<div class="sceditor-resize-cover" />'),
524 				startX		= 0,
525 				startY		= 0,
526 				startWidth	= 0,
527 				startHeight	= 0,
528 				origWidth	= $editorContainer.width(),
529 				origHeight	= $editorContainer.height(),
530 				dragging	= false,
531 				minHeight, maxHeight, minWidth, maxWidth, mouseMoveFunc;
532 
533 			minHeight = (base.options.resizeMinHeight == null ?
534 					origHeight / 2 :
535 					base.options.resizeMinHeight);
536 
537 			maxHeight = (base.options.resizeMaxHeight == null ?
538 					origHeight * 2 :
539 					base.options.resizeMaxHeight);
540 
541 			minWidth = (base.options.resizeMinWidth == null ?
542 					origWidth / 2 :
543 					base.options.resizeMinWidth);
544 
545 			maxWidth = (base.options.resizeMaxWidth == null ?
546 					origWidth * 2 :
547 					base.options.resizeMaxWidth);
548 
549 			mouseMoveFunc = function (e) {
550 				var	newHeight = startHeight + (e.pageY - startY),
551 					newWidth  = startWidth  + (e.pageX - startX);
552 
553 				if (newWidth >= minWidth && (maxWidth < 0 || newWidth <= maxWidth))
554 					base.width(newWidth);
555 
556 				if (newHeight >= minHeight && (maxHeight < 0 || newHeight <= maxHeight))
557 					base.height(newHeight);
558 
559 				e.preventDefault();
560 			};
561 
562 			$editorContainer.append($grip);
563 			$editorContainer.append($cover.hide());
564 
565 			$grip.mousedown(function (e) {
566 				startX		= e.pageX;
567 				startY		= e.pageY;
568 				startWidth	= $editorContainer.width();
569 				startHeight	= $editorContainer.height();
570 				dragging	= true;
571 
572 				$editorContainer.addClass('resizing');
573 				$cover.show();
574 				$(document).bind('mousemove', mouseMoveFunc);
575 				e.preventDefault();
576 			});
577 
578 			$(document).mouseup(function (e) {
579 				if(!dragging)
580 					return;
581 
582 				dragging = false;
583 				$cover.hide();
584 
585 				$editorContainer.removeClass('resizing');
586 				$(document).unbind('mousemove', mouseMoveFunc);
587 				e.preventDefault();
588 			});
589 		};
